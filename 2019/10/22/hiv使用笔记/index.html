1.创建表
drop table if exists app.test;
create table if not exists app.test
(
dim_day_id string comment ‘统计日期’,
pop_vender_id string comment ‘商家id’,
user_gro_type string comment ‘用户群类型’,
user_id string comment ‘用户id’,
user_ord_lv_cd string comment ‘用户级别代码’
) PARTITIONED BY (dt string)

2.删除表
drop table if exists app.test;
3.插入数据
set mapred.output.compress=true;
set hive.exec.compress.output=true;
set mapred.output.compression.codec=com.hadoop.compression.lzo.LzopCodec ;
set io.compression.codecs=com.hadoop.compression.lzo.LzopCodec;
insert overwrite table app.test partition (dt=’2014-03-16’)
select tt.dim_day_id,tt.pop_vender_id,tt.user_gro_type,tt.user_id,tt.user_ord_lv_cd
from app.sources tt;
set mapred.output.compress=false;
set hive.exec.compress.output=false;
4.查看分区
show partitions app.test;
5.删除分区
alter table app.test drop if exists partition(dt=’2014-03-16’);
6.增加分区
alter table app.test add if exists partition(dt=’2014-03-17’) location ‘/app/dt=2014-03-17’;
7.表重命名
alter table app.test rename to app.test1;
8.load文件数据到hive表里
drop table if exists dev.a_03_market_coupon_cate_pici;
create table dev.a_03_market_coupon_cate_pici
(cate_cd string,
pici string)
ROW FORMAT SERDE ‘org.apache.hadoop.hive.serde2.lazy.LazySimpleSerDe’
WITH SERDEPROPERTIES (‘field.delim’=’\t’,’escape.delim’=’\n’,’serialization.null.format’=’’);
load data local inpath ‘/export/huzf/tmp.txt’ overwrite into table dev.a_03_market_coupon_cate_pici;
